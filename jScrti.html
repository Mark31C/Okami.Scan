import { getFirestore, collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.x.x/firebase-firestore.js"; import { getStorage, ref, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.x.x/firebase-storage.js"; const db = getFirestore(); const storage = getStorage(); window.subirCapituloCompleto = async () => { const numCap = document.getElementById('cap-numero').value; const mangaId = document.getElementById('manga-destino').value; const files = document.getElementById('cap-paginas').files; const progressBar = document.getElementById('upload-progress'); const statusText = document.getElementById('status-text'); if (!numCap || files.length === 0) { alert("Faltan datos o imágenes."); return; } progressBar.style.display = 'block'; const urlsPaginas = []; try { // Recorremos cada archivo seleccionado for (let i = 0; i < files.length; i++) { const file = files[i]; statusText.innerText = `Subiendo página ${i + 1} de ${files.length}...`; // Creamos una ruta organizada: manga/capitulo/nombre-archivo const storageRef = ref(storage, `mangas/${mangaId}/capitulo-${numCap}/${file.name}`); // Subida individual const snapshot = await uploadBytes(storageRef, file); const downloadURL = await getDownloadURL(snapshot.ref); urlsPaginas.push(downloadURL); // Guardamos la URL en nuestra lista // Actualizar progreso visual progressBar.value = ((i + 1) / files.length) * 100; } // Una vez subidas todas las fotos, guardamos la referencia en Firestore await addDoc(collection(db, "capitulos"), { mangaId: mangaId, numero: parseFloat(numCap), paginas: urlsPaginas, // Aquí va el array con todos los links fechaSubida: serverTimestamp() }); statusText.innerText = "¡Capítulo subido exitosamente!"; alert("Todo listo en la base de datos."); location.reload(); } catch (error) { console.error("Error en la subida:", error); statusText.innerText = "Error al subir el capítulo."; } };
