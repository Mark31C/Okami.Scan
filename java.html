import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/10.x.x/firebase-firestore.js"; import { getStorage, ref, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.x.x/firebase-storage.js"; const db = getFirestore(); const storage = getStorage(); // Función para mostrar/ocultar formularios según el botón presionado window.mostrarFormulario = (tipo) => { const form = document.getElementById('formulario-subida'); const title = document.getElementById('form-title'); form.style.display = 'block'; if (tipo === 'manga') { title.innerText = "Registrar Nuevo Manga"; // Aquí podrías limpiar o cambiar los inputs para el manga } else { title.innerText = "Subir Nuevo Capítulo"; // Aquí podrías añadir un selector para elegir a qué manga pertenece } }; // FUNCIÓN PRINCIPAL DE SUBIDA window.ejecutarSubidaFirebase = async () => { const nombre = document.getElementById('manga-nombre').value; const archivo = document.getElementById('manga-portada').files[0]; if (!nombre || !archivo) { alert("Por favor, completa todos los campos."); return; } try { console.log("Subiendo archivo..."); // 1. Subir imagen a Firebase Storage const storageRef = ref(storage, `portadas/${archivo.name}`); await uploadBytes(storageRef, archivo); const urlImagen = await getDownloadURL(storageRef); // 2. Guardar datos en Firestore await addDoc(collection(db, "mangas"), { titulo: nombre, portada: urlImagen, fechaCreacion: new Date() }); alert("¡Manga publicado con éxito en Okami Scan!"); location.reload(); // Recarga para ver los cambios } catch (error) { console.error("Error al subir:", error); alert("Hubo un error al subir el contenido."); } };
